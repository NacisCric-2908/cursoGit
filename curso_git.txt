# Instalacion de MAN para la documentacion
sudo pacman -S man-db man-pages less

# Instalar git
sudo pacman -S git

# Todos los comandos
git -h

# Documentacion de un comando
git help COMANDO

# Confirgurar usuario y nombre GLOBALES
git config --global user.name "NOMBRE"
git config --global user.email "EMAIL"
git config --list

# Inicializar repo
git init
git status

# Agregar archivo al Stash area
git add FILE or git add .

# Primer commit
git commit -m "MENSAJE"

# Descarta cambios en un archivo no commiteado
git checkout -- FILE

# Descarta todos los cambios en el directorio actual
git checkout -- .

# Restaurar un archivo a los cambios de commit (n) específico
git checkout HASH -- FILE or git checkout HEAD~n -- FILE

# Ver los commits con sus hashes completos
git log

# Formato más compacto
git log --oneline

# Ver los últimos N commits (ej: últimos 3)
git log -3
git log --oneline -5

# Moverse entre commits (solo vista, recuerda siempre retornar al ultimo commit con este mismo comando)
git reset --mixed HASH

# Retornar a un commit (CUIDADO: elimina los cambios)
git reset --hard HASH

# Ver TODOS los cambios
git reflog
